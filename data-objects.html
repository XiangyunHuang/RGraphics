3.2 日期

注意观察时间转化

Sys.Date()
#> [1] "2020-02-05"
Sys.time()
#> [1] "2020-02-05 13:15:01 CST"
c(Sys.time(), Sys.Date())
#> [1] "2020-02-05 13:15:01 CST" "1970-01-01 13:04:57 CST"
data.table::year(Sys.Date())
#> [1] 2020
data.table::year(Sys.time())
#> [1] 2020
data.table::year(c(Sys.time(), Sys.Date()))
#> [1] 2020 1970
x <- Sys.time()
class(x)
#> [1] "POSIXct" "POSIXt"
format(x, format = "%Y-%m-%d")
#> [1] "2020-02-05"
x <- c("2019-12-21", "2019/12/21")
data.table::year("2019-12-21")
#> [1] 2019
data.table::year("2019/12/21")
#> [1] 2019

但是，下面这样会报错

data.table::year(x)
#> Error in as.POSIXlt.character(x): character string is not in a standard unambiguous format

正确的姿势是首先将表示日期的字符串格式统一

gsub(pattern = "/", replacement = "-", x) %>% 
  data.table::year()
#> [1] 2019 2019

date-times 表示 POSIXct 和 POSIXlt 类型的日期对象

(x <- Sys.time())
#> [1] "2020-02-05 13:15:01 CST"
class(x)
#> [1] "POSIXct" "POSIXt"

data.table::second(x) # 取秒
#> [1] 1
format(x, format = "%S")
#> [1] "01"

data.table::minute(x) # 取分
#> [1] 15
format(x, format = "%M")
#> [1] "15"

data.table::hour(x) # 取时
#> [1] 13
format(x, format = "%H")
#> [1] "13"
data.table::yday(x) # 此刻在一年的第几天
#> [1] 36

data.table::wday(x) # 此刻在一周的第几天，星期日是第1天，星期六是第7天
#> [1] 4
data.table::mday(x) # 此刻在当月第几天
#> [1] 5
format(x, format = "%d")
#> [1] "05"

weekdays(x)
#> [1] "Wednesday"
weekdays(x, abbreviate = T)
#> [1] "Wed"

data.table::week(x) # 此刻在第几周
#> [1] 6
data.table::isoweek(x)
#> [1] 6

data.table::month(x) # 此刻在第几月
#> [1] 2
format(x, format = "%m")
#> [1] "02"

months(x)
#> [1] "February"
months(x, abbreviate = T)
#> [1] "Feb"

data.table::quarter(x) # 此刻在第几季度
#> [1] 1
quarters(x)
#> [1] "Q1"

data.table::year(x) # 取年
#> [1] 2020
format(x, format = "%Y")
#> [1] "2020"

format(difftime(Sys.time(), x, units = "secs"))
#> [1] "0.05605173 secs"

3.11 时间序列

Jeffrey A. Ryan 在开设了一门免费课程教大家如何在 R 语言中使用 xts 和 zoo 包操作时间序列数据15

xts (eXtensible Time Series) 扩展的 zoo 对象

xts(x = NULL,
    order.by = index(x),
    frequency = NULL,
    unique = TRUE,
    tzone = Sys.getenv("TZ"),
    ...)
library(zoo)
library(xts)
x = matrix(1:4, ncol = 2,nrow = 2)
idx <- as.Date(c("2018-01-01", "2019-12-12"))
# xts = matrix + index
xts(x, order.by = idx)
#>            [,1] [,2]
#> 2018-01-01    1    3
#> 2019-12-12    2    4
